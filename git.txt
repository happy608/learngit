


echo "# learngit" >> README.md
git init #本地初始化仓库
git add README.md  #添加文件并提交（本地仓库有了提交记录才能推送）
git commit -m "first commit"
git branch -M main #强制将当前分支（通常是 master）重命名为 main，-M 保证重命名在目标分支已存在时也能生效；
git remote add origin https://github.com/xx608/learngit.git  #把本地代码推送到远程的 main 分支（-u 表示关联本地和远程分支）
git push -u origin main   #把当前分支main推送到远程

git checkout -b dev #创建并切换成dev分支 相当于以下两条命令
$ git branch dev
$ git checkout dev
Switched to branch 'dev'

git switch -c dev #创建并切换成dev分支

git branch #查看当前分支
git branch  dev #创建dev分支
git checkout main #切换到main分支
git merge dev #合并dev到当前分支
git branch -d dev # 删除dev分支
